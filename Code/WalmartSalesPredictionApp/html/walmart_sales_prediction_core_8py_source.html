49def recreate_arima_model(params):
50 """
51 @brief Attempt to recreate an ARIMA model from parameters if pickle loading fails
52 @details This function serves as a fallback mechanism when model deserialization fails
53 due to version incompatibilities or missing dependencies
54 @param params Dictionary containing model parameters, must include 'order' key
55 @return ARIMA model object or None if recreation fails
56 @raises ValueError If parameters are not a dictionary or order is invalid
57 @warning This creates a dummy model and may not preserve original training data
58 """
59 # Validate input parameter type
60 if not isinstance(params, dict):
61 raise ValueError("Parameters must be a dictionary")
62
63 try:
64 # Extract ARIMA order parameters, use default if not provided
65 order = params.get('order', CONFIG['DEFAULT_ARIMA_ORDER'])
66
67 # Validate order tuple structure (p, d, q)
68 if not isinstance(order, tuple) or len(order) != 3:
69 raise ValueError("Order must be a tuple of length 3")
70
71 # Create dummy ARIMA model with specified order
72 # Note: This creates a model with dummy data, not the original training data
73 model = ARIMA(np.array([0]), order=order)
74 return model
75 except Exception as e:
76 # Log warning and return None on failure
77 warnings.warn(f"Failed to recreate ARIMA model: {str(e)}")

80def load_default_model(model_type):
81 """
82 @brief Load default model from models/default/ directory with improved error handling
83 @details Attempts to load pre-trained models using both joblib and pickle for compatibility
84 @param model_type Type of model to load (must be in MODEL_FILE_MAP)
85 @return Tuple of (model_object, error_message) where error_message is None on success
86 @raises ValueError If model_type is empty or invalid
87 @note Uses fallback loading mechanisms for cross-platform compatibility
88 """
89 # Validate input parameter
90 if not model_type:
91 raise ValueError("Model type cannot be empty")
92
93 # Check if model type is supported
94 if model_type not in CONFIG['MODEL_FILE_MAP']:
95 return None, f"Invalid model type: {model_type}"
96
97 # Construct file path using mapping
98 file_name = CONFIG['MODEL_FILE_MAP'][model_type]
99 model_path = f"{CONFIG['DEFAULT_MODEL_PATH']}{file_name}.pkl"
100
101 # Check if model file exists
102 if not os.path.exists(model_path):
103 return None, f"Default model not found at {model_path}"
104
105 try:
106 # First try joblib loading (preferred method)
107 try:
108 model = joblib.load(model_path)
109 return model, None
110 except Exception as joblib_error:
111 # If joblib fails, try pickle as fallback
112 try:
113 with open(model_path, 'rb') as file:
114 model = pickle.load(file)
115 return model, None
116 except Exception as pickle_error:
117 # Handle specific statsmodels compatibility issues
118 if model_type == "Auto ARIMA" and ("statsmodels" in str(joblib_error) or "statsmodels" in str(pickle_error)):
119 return None, "Error loading model. Please check the model file or try another model type."
120 # Raise generic error for other issues
121 raise Exception(f"Failed to load model: {str(joblib_error)}\n{str(pickle_error)}")
122 except Exception as e:
123 return None, f"Error loading model: {str(e)}"

162 return model, None
163 except Exception as pickle_error:
164 # Handle specific statsmodels compatibility issues
165 if model_type == "Auto ARIMA" and ("statsmodels" in str(joblib_error) or "statsmodels" in str(pickle_error)):
166 # Clean up temporary file before returning error
167 os.unlink(tmp_path)
168 return None, "Error loading model. Please check the model file or try another model type."
169 # Raise generic error for other loading failures
170 raise Exception(f"Failed to load model: {str(joblib_error)}\n{str(pickle_error)}")
171
172 except Exception as e:
173 # Clean up temporary file if any error occurs
174 if tmp_path:

176 os.unlink(tmp_path)

181def predict_next_4_weeks(model, model_type):
182 """
183 @brief Predict next 4 weeks of sales using the provided model
184 @details Generates predictions based on model type and creates corresponding date ranges
185 @param model Trained model object (ARIMA or Exponential Smoothing)
186 @param model_type Type of model being used for prediction routing
187 @return Tuple of (predictions_array, dates_list, error_message) where error_message is None on success
188 @raises ValueError If model is None or model_type is empty
189 @note Dates are generated starting from next week after current date
190 """
191 # Validate input parameters
192 if not model:
193 raise ValueError("Model cannot be None")
194
195 if not model_type:
196 raise ValueError("Model type cannot be empty")
197
198 # Generate dates for next 4 weeks starting from next week
199 today = datetime.now()
200 dates = [today + timedelta(weeks=i) for i in range(1, CONFIG['PREDICTION_PERIODS'] + 1)]
201
202 try:
203 # Map model type to functional type for prediction routing
204 functional_model_type = CONFIG['MODEL_FUNC_MAP'].get(model_type)
205 if not functional_model_type:
206 raise ValueError(f"Unknown model type: {model_type}")
207
208 # Route to appropriate prediction method based on model type
209 if functional_model_type == "Auto ARIMA":
210 # Use ARIMA-specific prediction method
211 predictions = model.predict(n_periods=CONFIG['PREDICTION_PERIODS'])
212 elif functional_model_type == "Exponential Smoothing (Holt-Winters)":
213 # Use Exponential Smoothing-specific forecast method
214 predictions = model.forecast(CONFIG['PREDICTION_PERIODS'])
215 else:
216 # Unknown model type error
217 raise ValueError(f"Unknown model type: {functional_model_type}")
218
219 return predictions, dates, None
220 except Exception as e:
221 # Return error information on prediction failure
222 return None, None, f"Error generating predictions: {str(e)}"
